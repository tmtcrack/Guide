#!/bin/bash

# Prompt the user for the MySQL root password
read -sp "Enter MySQL root password: " mysql_root_password
echo

# Log in to MySQL as root
mysql -u root -p$mysql_root_password <<EOF

# Create a new MySQL user for zabbix
CREATE USER 'zabbix'@'localhost' IDENTIFIED BY 'ZABBIX@1983zxcv';

# Create a new MySQL database for zabbix
CREATE DATABASE zabbix CHARACTER SET utf8 COLLATE utf8_bin;

# Grant privileges for the zabbix user on the zabbix database
GRANT ALL PRIVILEGES ON zabbix.* TO 'zabbix'@'localhost';

# Exit MySQL
exit

EOF
